def hanoi(n, origem, auxiliar, destino):
    # Caso base: só 1 disco → mova diretamente
    if n == 1:
        print(f"Mover disco 1: {origem} → {destino}")
        return

    # Passo 1: mova os N-1 discos de cima para o auxiliar
    hanoi(n - 1, origem, destino, auxiliar)

    # Passo 2: mova o disco maior para o destino
    print(f"Mover disco {n}: {origem} → {destino}")

    # Passo 3: mova os N-1 discos do auxiliar para o destino
    hanoi(n - 1, auxiliar, origem, destino)

# Chamada para 3 discos
hanoi(3, 'A', 'B', 'C')
